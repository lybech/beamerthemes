\documentclass[10pt,xcolor={dvipsnames}]{beamer}

%%% option passed to the outer theme
%progressstyle=fixedCircCnt,   % fixedCircCnt, movingCircCnt (moving is deault)
\usetheme[]{Hest}
  
\setbeamercolor{Hest}{fg=NavyBlue!20,bg=NavyBlue} % Change the bar colors:
\setbeamercolor{structure}{fg=NavyBlue} % Change the color of the structural elements:
\setbeamercolor{frametitle}{fg=black!5} % Change the frame title text color:
\setbeamercolor{normal text}{fg=black!75,bg=gray!5} % Change the normal text colors:
\setbeamercolor{block title}{use=Hest,bg=Hest.fg,fg=black!90} %% Change the block title colors
